<script>
let hHP = 100, mHP = 100;
let questionBank = [];
let currentIndex = 0;

// 1. 生成 100 題完全不同的題目
function generateQuestions() {
  const types = [
    { title: "關於營養：", good: ["多吃蔬菜", "喝足夠的水", "吃天然原型食物", "補充維生素D", "攝取優質蛋白"], bad: ["常喝含糖飲料", "吃加工肉品", "天天吃油炸物", "節食不吃澱粉", "攝取過多反式脂肪"] },
    { title: "關於生活習慣：", good: ["每天睡足7小時", "飯後散步", "定時運動", "睡前不看手機", "保持心情愉快"], bad: ["長期熬夜", "久坐不動", "抽菸酗酒", "過度使用3C", "不吃早餐"] },
    { title: "關於身體警訊：", good: ["定期健康檢查", "注意體重變化", "觀察排便習慣", "保持牙齒清潔", "有病痛即就醫"], bad: ["亂吃止痛藥", "無視長期疲勞", "信偏方不就醫", "不愛曬太陽", "過度勞累"] }
  ];

  for (let i = 1; i <= 100; i++) {
    const type = types[Math.floor(Math.random() * types.length)];
    const isGoodQ = Math.random() > 0.5;
    const correct = isGoodQ ? type.good[Math.floor(Math.random() * 5)] : type.bad[Math.floor(Math.random() * 5)];
    
    // 隨機選三個錯誤答案
    let wrongs = [];
    const pool = isGoodQ ? type.bad : type.good;
    wrongs = [...pool].sort(() => 0.5 - Math.random()).slice(0, 3);

    const allOptions = [correct, ...wrongs].sort(() => 0.5 - Math.random());
    
    questionBank.push({
      q: `第 ${i} 題 | ${type.title} 下列哪項行為「${isGoodQ ? '最健康' : '最不健康'}」？`,
      options: allOptions,
      answer: allOptions.indexOf(correct)
    });
  }
}

// 2. 遊戲控制
function startGame() {
  document.getElementById("startNotice").style.display = "none";
  const bgm = document.getElementById("bgm");
  bgm.volume = 0.8; // 調大背景音樂
  bgm.play();
  loadQuestion();
}

function loadQuestion() {
  if (currentIndex >= 100) {
    alert("你完成了 100 題挑戰！你是真正的健康之王！");
    location.reload();
    return;
  }

  const data = questionBank[currentIndex];
  document.getElementById("question").innerText = data.q;
  const choicesDiv = document.getElementById("choices");
  choicesDiv.innerHTML = "";

  data.options.forEach((opt, i) => {
    const btn = document.createElement("button");
    btn.innerText = opt;
    btn.onclick = () => handleAnswer(i === data.answer);
    choicesDiv.appendChild(btn);
  });
}

function handleAnswer(isCorrect) {
  if (isCorrect) {
    fire(true);
  } else {
    fire(false);
  }
  currentIndex++;
  setTimeout(loadQuestion, 800);
}

function fire(isHero) {
  document.getElementById("shootSound").play();
  const bullet = document.createElement("div");
  bullet.className = "bullet";
  document.body.appendChild(bullet);

  const heroEl = document.getElementById("hero");
  const monsterEl = document.getElementById("monster");
  
  let startX = isHero ? 400 : window.innerWidth - 400;
  let startY = window.innerHeight / 2;
  bullet.style.left = startX + "px";
  bullet.style.top = startY + "px";

  let x = startX;
  const move = setInterval(() => {
    x += isHero ? 30 : -30;
    bullet.style.left = x + "px";

    // 碰撞偵測
    if (x > window.innerWidth - 400 || x < 400) {
      clearInterval(move);
      bullet.remove();
      document.getElementById("hitSound").play();
      
      if (isHero) {
        mHP -= 10;
        monsterEl.classList.add("hit-flash");
        setTimeout(() => monsterEl.classList.remove("hit-flash"), 150);
      } else {
        hHP -= 20;
        heroEl.classList.add("hit-flash");
        setTimeout(() => heroEl.classList.remove("hit-flash"), 150);
      }
      
      updateUI();
    }
  }, 20);
}

function updateUI() {
  document.getElementById("heroHP").style.width = hHP + "%";
  document.getElementById("monsterHP").style.width = mHP + "%";

  if (hHP <= 0) { alert("挑戰失敗！健康需要持之以恆。"); location.reload(); }
  if (mHP <= 0) { 
    alert("擊敗了病魔！繼續挑戰下一關！"); 
    mHP = 100; // 重置怪物血量繼續答題
    updateUI();
  }
}

// 初始化題目
generateQuestions();
